#
# Requires Python 3.12 or 3.13  (faster-whisper / ctranslate2 have no wheels for 3.14).
# Also requires the ffmpeg BINARY on PATH (not a pip package):  brew install ffmpeg
#
#   python3.12 -m venv .venv
#   source .venv/bin/activate
#   pip install -r requirements.txt        # deps + the lingus package itself
#
# Then, with OPENAI_API_KEY set in .env for the hosted generator:
#   python -m lingus.app --platform youtube --video G43NInZfoPE --web
#   # open http://127.0.0.1:8080
#
# Dev tooling (tests + lint) is not needed to run; install it from pyproject when
# needed:  pip install -e ".[dev]"
